4.1 Annotations

  • title

library(gt)
tbl <- gtcars |> 
  dplyr::count(mfr, sort = TRUE) |> 
  dplyr::slice(1:5) |> 
  gt(id = 'my_tbl', rowname_col = 'mfr') |> 
  opt_stylize(style = 3) |> 
  cols_label(n = 'Number of cars') 

tbl |> as_raw_html()
Number of cars
Ferrari 9
Audi 5
BMW 5
Aston Martin 4
Porsche 4
 

tbl |> 
  opt_css('
   #my_tbl thead, tbody, tfoot, tr, td, th {
      border-style: none;
    }
   ') |> 
  as_raw_html()
Number of cars
Ferrari 9
Audi 5
BMW 5
Aston Martin 4
Porsche 4
